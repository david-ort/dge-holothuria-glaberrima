#!/bin/sh
#SBATCH --time 02:00:00
#SBATCH -p debug -n 8
#SBATCH --job-name=bowtie
#SBATCH --output=bowtie.out
#SBATCH --error=bowtie.err
#SBATCH --mail-type=ALL
#SBATCH --mail-user=arciladk@gmail.com



module load bowtie/1.0.0  

#########################
# Build Index
#########################

bowtie-build -f [options]* <reference> <nameofIndex>

bowtie-build -f AstyanaxMexicanus.fasta AstyanaxMexicanus

bowtie-build -f Stygicthys.fasta Stygicthys


#########################
# Bowtie alignments for mate-paired reads
#########################

#Astyanax#

bowtie ~/FILEPATH/AstyanaxMexicanus -f -n 2 -m 300  --all --best --strata  --chunkmbs 512  -q ~/FILEPATH/mRNASeqName.raw -S ~/FILEPATH/AstyanaxMexicanus_Surface_Output.sam
bowtie ~/FILEPATH/AstyanaxMexicanus -f -n 2 -m 300 --all --best --strata  --chunkmbs 512  -q ~/FILEPATH/mRNASeqName.raw -S ~/FILEPATH/AstyanaxMexicanus_Cave_Output.sam

#Stygicthys#

bowtie ~/FILEPATH/Stygicthys -f -n 2 -m 300 --all --best --strata --chunkmbs 512  -q ~/FILEPATH/mRNASeqName_Ind1.raw -S ~/FILEPATH/Stygicthys_Cave_Ind1_Output.sam
bowtie ~/FILEPATH/Stygicthys -f -n 2 -m 300 --all --best --strata --chunkmbs 512  -q ~/FILEPATH/mRNASeqName_Ind2.raw -S ~/FILEPATH/Stygicthys_Cave_Ind2_Output.sam
bowtie ~/FILEPATH/Stygicthys -f -n 2 -m 300 --all --best --strata --chunkmbs 512  -q ~/FILEPATH/mRNASeqName_Ind3.raw -S ~/FILEPATH/Stygicthys_Cave_Ind3_Output.sam




#the -f flag tells botwie that reference files are Fasta
#the -n 2 flag denotes the maximum number of mistmatches permitted in the seed (default value 2)
#the -m flag tells botwie to supress all alignments if >300
#the --strata flag tells botwie hits in sub-optimal strata aren't reported (requires --best)
#the ---chunkmbsflag tells botwie the max megabytes of RAMS for best-first search frames
#the -q flag tells botwie that input files are FASTQ .fq
#the -S flag tells botwie to write the output in SAM format
